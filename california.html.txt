    // ビューの切り替え (タイムライン ⇔ 詳細リスト)
    function switchView(view) {
      const btnTimeline = document.getElementById('btn-timeline');
      const btnList = document.getElementById('btn-list');
      const viewTimeline = document.getElementById('view-timeline');
      const viewList = document.getElementById('view-list');

      if (view === 'timeline') {
        btnTimeline.className = 'px-3 py-1.5 rounded text-xs font-semibold bg-white text-slate-800 shadow-sm transition';
        btnList.className = 'px-3 py-1.5 rounded text-xs font-semibold text-white hover:bg-white/10 transition';
        viewTimeline.classList.remove('hidden');
        viewList.classList.add('hidden');
      } else {
        btnTimeline.className = 'px-3 py-1.5 rounded text-xs font-semibold text-white hover:bg-white/10 transition';
        btnList.className = 'px-3 py-1.5 rounded text-xs font-semibold bg-white text-slate-800 shadow-sm transition';
        viewTimeline.classList.add('hidden');
        viewList.classList.remove('hidden');
      }
    }

    // タイムライン描画処理
    function renderTimeline() {
      const container = document.getElementById('timeline-rows');
      container.innerHTML = '';

      scheduleData.forEach(day => {
        // 各日の行コンテナ
        const row = document.createElement('div');
        row.className = 'grid grid-cols-[110px_repeat(16,1fr)] py-3.5 items-center relative min-h-[72px] transition hover:bg-slate-50/50';

        // 日付・曜日列
        const dateCol = document.createElement('div');
        const dayClass = day.isWeekend ? 'text-red-500 font-bold' : 'text-slate-600';
        dateCol.className = 'text-left pl-2 select-none z-10';
        dateCol.innerHTML = `
          <div class="text-sm font-black text-slate-800 flex items-center gap-1">
            ${day.date} <span class="${dayClass}">(${day.dayOfWeek})</span>
          </div>
          <span class="inline-block mt-1 px-1.5 py-0.5 rounded text-[10px] font-bold ${day.tagColor}">
            ${day.tag}
          </span>
        `;
        row.appendChild(dateCol);

        // 背景のダッシュライン (グリッド線)
        const bgGrid = document.createElement('div');
        bgGrid.className = 'absolute inset-y-0 left-[110px] right-0 grid grid-cols-16 pointer-events-none select-none';
        for (let i = 0; i < 16; i++) {
          const gridLine = document.createElement('div');
          gridLine.className = 'border-r border-dashed border-slate-100 h-full';
          bgGrid.appendChild(gridLine);
        }
        row.appendChild(bgGrid);

        // 各イベントの配置
        day.events.forEach(event => {
          // グリッドスパンの計算
          // 8:00がカラム2、24:00がカラム18
          const startCol = event.start - 8 + 2;
          const endCol = event.end - 8 + 2;

          const eventBlock = document.createElement('div');
          eventBlock.className = `mx-1 p-1.5 rounded-lg border text-xs shadow-sm flex flex-col justify-center cursor-pointer transition transform hover:scale-[1.01] hover:shadow-md z-10 overflow-hidden ${event.color}`;
          eventBlock.style.gridColumn = `${startCol} / ${endCol}`;
          eventBlock.title = `${event.title}\n時間: ${event.start}:00 〜 ${event.end}:00\n詳細: ${event.desc}`;
          
          eventBlock.innerHTML = `
            <div class="font-extrabold truncate">${event.title}</div>
            <div class="text-[9px] opacity-75 truncate">${event.start}:00 - ${event.end}:00</div>
          `;
          row.appendChild(eventBlock);
        });

        container.appendChild(row);
      });
    }
